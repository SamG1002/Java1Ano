	public static long fatorial (int base) {
		long fatorial=1;
		if (base<0) {
			System.out.println("Não é possível fatorar um número negativo");
		} else {
			while (base>1) {
				fatorial= fatorial*base;
				base--;
			}
		}
		return fatorial;
	}

	public static void potenciacao () {
